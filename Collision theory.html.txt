    <script>
        const canvas = document.getElementById('simCanvas');
        const ctx = canvas.getContext('2d');
        
        let particles = [];
        let temp = 298;
        let conc = 30;
        let productsFormed = 0;
        let lastTimestamp = 0;
        let reactionEventsInLastSecond = 0;
        let timer = 0;

        const EA_THRESHOLD = 15; // 固定活化能門檻

        class Particle {
            constructor(type) {
                this.type = type; // 'A', 'B', 'C'
                this.init();
            }

            init() {
                this.radius = 6;
                this.x = Math.random() * (canvas.width - 20) + 10;
                this.y = Math.random() * (canvas.height - 20) + 10;
                this.life = 1.0; // 用於 C 生成物的生命週期（之後會分解回 A 和 B）
                
                const angle = Math.random() * Math.PI * 2;
                const speed = (Math.random() * 1.5 + 0.5);
                this.vx = Math.cos(angle) * speed;
                this.vy = Math.sin(angle) * speed;
            }

            update() {
                // 速度受溫度影響 (v ∝ √T)
                const tempFactor = Math.sqrt(temp / 298);
                this.x += this.vx * tempFactor;
                this.y += this.vy * tempFactor;

                // 邊界碰撞
                if (this.x < this.radius || this.x > canvas.width - this.radius) this.vx *= -1;
                if (this.y < this.radius || this.y > canvas.height - this.radius) this.vy *= -1;

                // 如果是生成物 C，慢慢變回 A 和 B 以保持濃度平衡
                if (this.type === 'C') {
                    this.life -= 0.005;
                    if (this.life <= 0) {
                        this.type = Math.random() > 0.5 ? 'A' : 'B';
                        this.life = 1.0;
                    }
                }
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                
                if (this.type === 'A') {
                    ctx.fillStyle = '#60a5fa'; // Blue
                    ctx.shadowBlur = 0;
                } else if (this.type === 'B') {
                    ctx.fillStyle = '#fb7185'; // Rose
                    ctx.shadowBlur = 0;
                } else {
                    ctx.fillStyle = '#10b981'; // Emerald (Product C)
                    ctx.shadowBlur = 15 * this.life;
                    ctx.shadowColor = '#10b981';
                }
                
                ctx.fill();
                ctx.closePath();
                ctx.shadowBlur = 0;
            }
        }

        function setupParticles() {
            particles = [];
            const count = Math.floor(conc * 0.8);
            for (let i = 0; i < count; i++) {
                particles.push(new Particle(i % 2 === 0 ? 'A' : 'B'));
            }
        }

        function checkCollisions() {
            for (let i = 0; i < particles.length; i++) {
                for (let j = i + 1; j < particles.length; j++) {
                    let p1 = particles[i];
                    let p2 = particles[j];

                    // 只考慮 A 和 B 的反應
                    if ((p1.type === 'A' && p2.type === 'B') || (p1.type === 'B' && p2.type === 'A')) {
                        let dx = p1.x - p2.x;
                        let dy = p1.y - p2.y;
                        let dist = Math.sqrt(dx*dx + dy*dy);

                        if (dist < p1.radius + p2.radius) {
                            // 計算相對速度 (能量)
                            let relVx = p1.vx - p2.vx;
                            let relVy = p1.vy - p2.vy;
                            let collisionEnergy = (relVx*relVx + relVy*relVy) * (temp / 200);

                            // 有效碰撞判斷
                            if (collisionEnergy > EA_THRESHOLD) {
                                p1.type = 'C';
                                p1.life = 1.0;
                                p2.type = 'C';
                                p2.life = 1.0;
                                
                                productsFormed++;
                                reactionEventsInLastSecond++;
                                document.getElementById('productCount').innerText = productsFormed;
                            }

                            // 物理彈開
                            p1.vx *= -1; p1.vy *= -1;
                            p2.vx *= -1; p2.vy *= -1;
                        }
                    }
                }
            }
        }

        function animate(timestamp) {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // 粒子更新與繪製
            particles.forEach(p => {
                p.update();
                p.draw();
            });

            checkCollisions();

            // 反應速率計算 (每秒更新一次)
            if (timestamp - timer > 1000) {
                const rate = reactionEventsInLastSecond;
                const barWidth = Math.min(rate * 5, 100);
                document.getElementById('rateBar').style.width = barWidth + "%";
                document.getElementById('rateLabel').innerText = `RATE: ${rate.toFixed(1)} units/s`;
                reactionEventsInLastSecond = 0;
                timer = timestamp;
            }

            requestAnimationFrame(animate);
        }

        // 滑桿監聽
        document.getElementById('tempRange').addEventListener('input', (e) => {
            temp = e.target.value;
            document.getElementById('tempDisp').innerText = temp + " K";
        });

        document.getElementById('concRange').addEventListener('input', (e) => {
            conc = e.target.value;
            document.getElementById('concDisp').innerText = conc + " %";
            setupParticles();
        });

        function resetSim() {
            productsFormed = 0;
            document.getElementById('productCount').innerText = "0";
            setupParticles();
        }

        window.onload = () => {
            canvas.width = canvas.offsetWidth;
            canvas.height = canvas.offsetHeight;
            setupParticles();
            requestAnimationFrame(animate);
            lucide.createIcons();
        };

        // 視窗調整大小
        window.onresize = () => {
            canvas.width = canvas.offsetWidth;
            canvas.height = canvas.offsetHeight;
        };
    </script>
